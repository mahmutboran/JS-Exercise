Exercise-2: 

Write a JavaScript program to get the length of a JavaScript object.

Solute:

            var student = {
                name : "David Rayy",
                sclass : "VI",
                rollno : 12 };
            
            Object.myLenght = function(myFunc){
                var count = 0;
                for (const key in myFunc) {
                    if (myFunc.hasOwnProperty(key)) {
                        count++;
                    }
                }
                return count
            }
            console.log(Object.myLenght(student))
